------ instalar angular --------
npm install -g @angular/cli@16
ng v

------ criando o projeto -------
ng new nome do aplicativo
ng serve    //Inicia o servidor local e atualiza automaticamente ao salvar mudanças.

------ crinado componetes/serviços -------
ng generate component nome || ng g c nome
ng generate service nome   || ng g s nome

npm start  //compilar e execultar
